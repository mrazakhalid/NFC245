// Step 1: Parse the pasted URL and extract the Place ID/CID async function handleUrlParsing() { const rawUrl = mapsUrlInput.value.trim(); if (!rawUrl) { showStatus("Please paste a valid Google Maps link first.", "error"); return; } showStatus("🔍 Analyzing Google Maps URL...", "loading"); try { let resolvedUrl = rawUrl; // Resolve mobile short URLs (goo.gl, g.page, app.goo.gl, googleusercontent) if (rawUrl.includes("goo.gl") || rawUrl.includes("g.page") || rawUrl.includes("app.goo.gl") || rawUrl.includes("googleusercontent.com")) { // Using corsproxy.io with redirect=manual mode to extract the target location header safely const proxyUrl = `https://corsproxy.io/?url=${encodeURIComponent(rawUrl)}`; const response = await fetch(proxyUrl, { redirect: 'manual' }); // Read response headers or text to catch where Google is sending us if (response.ok || response.status === 302 || response.status === 301) { const locationHeader = response.headers.get('location'); if (locationHeader) { resolvedUrl = locationHeader; } else { // Fallback: Parse the HTML text if the proxy embedded the destination inside the body const text = await response.text(); const urlMatch = text.match(/href="([^"]+)"/) || text.match(/window\.location\.replace\("([^"]+)"\)/); if (urlMatch) { resolvedUrl = urlMatch[1]; } else { // If the proxy resolved it directly to the landing page, read its response URL resolvedUrl = response.url; } } } else { throw new Error("Could not expand the short link. Try copying the full link from your browser search bar."); } } const decodedUrl = decodeURIComponent(resolvedUrl); // Match 1: Extract standard ChIJ Place ID const chijMatch = decodedUrl.match(/(ChIJ[a-zA-Z0-9_-]{23})/); // Match 2: Extract CID hex string patterns (e.g. !1s0x...:0x...) const hexMatch = decodedUrl.match(/!1s(0x[0-9a-fA-F]+:0x[0-9a-fA-F]+)/); if (chijMatch) { const placeId = chijMatch[1]; targetReviewUrl = `https://search.google.com/local/writereview?placeid=${placeId}`; } else if (hexMatch) { const cidHex = hexMatch[1]; targetReviewUrl = `https://search.google.com/local/writereview?fid=${cidHex}`; } else { throw new Error("Could not find a valid Google Business identifier. Ensure you are sharing a specific business location."); } // Success Transition stepState = 'write'; processBtn.innerText = "⚡ Write to NFC Tag"; processBtn.style.backgroundColor = "#059669"; // Change color to Success Green showStatus("Link processed successfully! Ready to write to NFC.", "success"); } catch (error) { showStatus(`❌ Processing Error: ${error.message}`, "error"); stepState = 'parse'; processBtn.innerText = "Set Up Tag Link"; processBtn.style.backgroundColor = ''; } }
