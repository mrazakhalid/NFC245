// Programmatic short-link expansion and hex extraction async function handleLinkProcess() { const rawUrl = mapsUrlInput.value.trim(); if (!rawUrl) { showStatus("Please paste a valid Google Maps link first.", "error"); return; } showStatus("🔍 Expanding short link via HTTP...", "loading"); try { let decodedUrl = decodeURIComponent(rawUrl); // 1. Detect if it is a short link const isShortLink = rawUrl.includes("goo.gl") || rawUrl.includes("g.page") || rawUrl.includes("app.goo.gl") || rawUrl.includes("googleusercontent.com"); if (isShortLink) { // 2. Make an HTTP request via AllOrigins proxy to bypass browser CORS blocks const proxyUrl = `https://api.allorigins.win/get?url=${encodeURIComponent(rawUrl)}`; const response = await fetch(proxyUrl); if (!response.ok) { throw new Error("HTTP request failed. Could not reach proxy server."); } const data = await response.json(); if (data && data.contents) { // This raw HTML string contains the expanded long URL Google redirected to decodedUrl = decodeURIComponent(data.contents); } else { throw new Error("Failed to read redirected URL payload."); } } // 3. Locate the hexadecimal code (CID) embedded inside the string (!1s0x...:0x...) // Matches: 0x[hex-string]:0x[hex-string] const hexMatch = decodedUrl.match(/(0x[0-9a-fA-F]+:0x[0-9a-fA-F]+)/); // Fallback: Check for standard ChIJ Place ID const chijMatch = decodedUrl.match(/(ChIJ[a-zA-Z0-9_-]{23})/); if (hexMatch) { // Step 3: Format Your Clean Place ID/CID Long Link using the "fid" parameter const cidHex = hexMatch[1]; targetReviewUrl = `https://search.google.com/local/writereview?fid=${cidHex}`; setReadyToWriteState("Short link successfully expanded and parsed!"); } else if (chijMatch) { const placeId = chijMatch[1]; targetReviewUrl = `https://search.google.com/local/writereview?placeid=${placeId}`; setReadyToWriteState("Short link successfully expanded and parsed!"); } else { throw new Error("Could not find a valid hexadecimal CID (0x...:0x...) or Place ID in the expanded URL."); } } catch (error) { showStatus(`❌ ${error.message}`, "error"); resetAppState(); if (currentTab === 'link') actionBtn.disabled = false; } }
