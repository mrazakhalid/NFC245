// Step 1: Parse the pasted URL and extract the Place ID/CID async function handleUrlParsing() { const rawUrl = mapsUrlInput.value.trim(); if (!rawUrl) { showStatus("Please paste a valid Google Maps link first.", "error"); return; } showStatus("🔍 Analyzing Google Maps URL...", "loading"); try { let resolvedUrl = rawUrl; // Resolve mobile short URLs (goo.gl, g.page, app.goo.gl, googleusercontent) if (rawUrl.includes("goo.gl") || rawUrl.includes("g.page") || rawUrl.includes("app.goo.gl") || rawUrl.includes("googleusercontent.com")) { // Corrected proxy format: prepend https://corsproxy.io/?url= const proxyUrl = `https://corsproxy.io/?url=${encodeURIComponent(rawUrl)}`; const response = await fetch(proxyUrl); if (response.ok) { resolvedUrl = response.url; // Extract target URL from the proxy redirect header } else { throw new Error("Could not reach Google servers to expand the short link. Try pasting the full browser link instead."); } } const decodedUrl = decodeURIComponent(resolvedUrl); // Match 1: Extract standard ChIJ Place ID const chijMatch = decodedUrl.match(/(ChIJ[a-zA-Z0-9_-]{23})/); // Match 2: Extract CID hex string patterns (e.g. !1s0x...:0x...) const hexMatch = decodedUrl.match(/!1s(0x[0-9a-fA-F]+:0x[0-9a-fA-F]+)/); if (chijMatch) { const placeId = chijMatch[1]; targetReviewUrl = `https://search.google.com/local/writereview?placeid=${placeId}`; } else if (hexMatch) { const cidHex = hexMatch[1]; targetReviewUrl = `https://search.google.com/local/writereview?fid=${cidHex}`; } else { throw new Error("Could not find a valid Google Business identifier inside that link. Please make sure you are sharing a specific business location."); } // Success Transition stepState = 'write'; processBtn.innerText = "⚡ Write to NFC Tag"; processBtn.style.backgroundColor = "#059669"; // Change color to Success Green showStatus("Link processed successfully! Ready to write to NFC.", "success"); } catch (error) { showStatus(`❌ Processing Error: ${error.message}`, "error"); stepState = 'parse'; processBtn.innerText = "Set Up Tag Link"; processBtn.style.backgroundColor = ''; } }
